    isMultiBranch = prefs.getInt('multiBranch');
    userBranchId = prefs.getInt('branchId');
    userDepartmentId = prefs.getInt('departmentId');
    selectedBranchId = userBranchId;
    if (kDebugMode) {
      print('➡️ isPickupTeam $isPickupTeam');
      print('➡️ isMultiBranch: $isMultiBranch');
      print('➡️ userBranchId: $userBranchId');

        Navigator.pushReplacement(
          context,
          MaterialPageRoute(builder: (context) => const PickHomeScreen()),
        );
      });
      return;
    }
    if (userDepartmentId == 7) {
      WidgetsBinding.instance.addPostFrameCallback((_) {
        Navigator.pushReplacement(
          context,
          MaterialPageRoute(builder: (context) => const DriverHomeScreen()),
        );
      });

    setState(() {

              .toList();

      selectedFilter = filterOptions.firstWhere(
        (f) => f.key == (isPickup ? 0 : 0),

      groupByOptions = [
        {"id": "Invoice", "name": "Invoice"},
        {"id": "Trip", "name": "Trip"},
        {"id": "Item", "name": "Item"},
      ];

    /*setState(() {

  Future<void> fetchInvoices() async {
    setState(() => isLoading = true);
    final newInvoices = await _invoiceService.getInvoices(
      filter: selectedFilter?.key.toString() ?? '0',
      page: currentPage,
      isPickup: isPickupTeam == 1,
      isMultiBranch: isMultiBranch,
      userBranchId: userBranchId,
      search: _searchController.text,
      groupBy: selectedGroup,
    );
    // invoices.addAll(newInvoices);
    for (var invoice in newInvoices) {
      if (!invoices.any((inv) => inv.id == invoice.id)) {
        invoices.add(invoice);
      }
    }
    setState(() {
      isLoading = false;
    });
  }

  /*void onSearchChanged(String query) {
    currentPage = 1;
    invoices.clear();
    fetchInvoices();
  }*/
  void onSearchChanged(String query) {
    setState(() {
      currentPage = 1;
      invoices.clear();
    });
    fetchInvoices();
  }

  void openInvoiceDet({Invoice? invoice}) async {
    await Navigator.push(
      context,
      MaterialPageRoute(builder: (_) => InvoiceDet(invoice: invoice)),
    );
    fetchInvoices();
  }

      body: Stack(
        children: [
          backgroundAllScreen(),
          Column(
            children: [
              const SizedBox(height: 80),
              if (filterOptions.isNotEmpty)

                  padding: const EdgeInsets.fromLTRB(16, 2, 16, 4),
                  child: CustomDropdown<FilterOption>(
                    label: "Filter by Status",
                    items: filterOptions,
                    selectedItem: selectedFilter,
                    onChanged: (option) {
                      if (option != null) {
                        setState(() {
                          selectedFilter = option;
                          currentPage = 1;
                          invoices.clear();
                        });
                        fetchInvoices();
                      }
                    },
                    getLabel: (option) => option.label,
                  ),

              // Show branch dropdown if multi-branch

                Padding(
                  padding: const EdgeInsets.fromLTRB(16, 8, 16, 8),
                  child: CustomDropdown<Branch>(
                    label: "Select Branch",
                    items: branchesDrop,
                    selectedItem: branchesDrop.firstWhere(
                      (b) => b.id == selectedBranchId,
                      orElse: () => Branch(id: 0, name: 'Select'),
                    ),
                    onChanged: (branch) {
                      if (branch != null) {
                        setState(() {
                          userBranchId = branch.id;
                          selectedBranchId = branch.id;
                          currentPage = 1;
                          invoices.clear();
                        });
                        fetchInvoices();
                      }

                  ),

                    label: "Group By",

                        groupByOptions.map((e) => e['id'] as String).toList(),
                    selectedItem: selectedGroup,
                    onChanged: (value) {
                      if (value != null) {
                        setState(() {
                          selectedGroup = value;
                          currentPage = 1;
                          invoices.clear();
                        });
                        fetchInvoices();
                      }
                    },

                      final item = groupByOptions.firstWhere(
